### クイックソート

```zig
pub fn quicksort() void {
    var array = [_]i32{ 4, 6, 1, 3, 9, 2, 1 };

    for (array) |_, i| {
        var j: usize = 0;
        while (j < array.len - i - 1) : (j += 1) {
            if (array[j] > array[j + 1]) {
                // swap
                const temp = array[j];
                array[j] = array[j+1];
                array[j+1] = temp;
            }
        }
    }

    // 出力
    for (array) | v | {
        std.debug.print("{d}\n", .{ v });
    }
}
```
